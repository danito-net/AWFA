# users

CREATE TABLE IF NOT EXISTS users (
       email TEXT,
       phone TEXT,
       fullname TEXT,
       nickname TEXT,
       password TEXT,
       device_id TEXT,
       device_status INT,
       attache_locale TEXT,
       attache_gender TEXT,
       attache_status INT,
       created TIMESTAMP,
       last_edited TIMESTAMP,
       notes TEXT,
       status INT,
       user_id UUID,
PRIMARY KEY ((email), user_id))
WITH CLUSTERING ORDER BY (user_id DESC);

CREATE TABLE IF NOT EXISTS devices (
       serial TEXT,
       description TEXT,
       created TIMESTAMP,
       activated TIMESTAMP,
       last_edited TIMESTAMP,
       notes TEXT,
       status INT,
       device_id UUID,
PRIMARY KEY ((serial), device_id))
WITH CLUSTERING ORDER BY (device_id DESC);

CREATE TABLE IF NOT EXISTS connections (
       connection_id TIMEUUID,
       user_id TEXT,
       device_id TEXT,
       started TIMESTAMP,
       finished TIMESTAMP,
       latitude TEXT,
       longitude TEXT,
       altitude TEXT,
       location TEXT,
       notes TEXT,
       status INT,
PRIMARY KEY ((user_id), connection_id))
WITH CLUSTERING ORDER BY (connection_id DESC);

CREATE TABLE IF NOT EXISTS sensors (
       device_id TEXT,
       latitude TEXT,
       longitude TEXT,
       altitude TEXT,
       temperature TEXT,
       humidity TEXT,
       pressure TEXT,
       light_intensity TEXT,
       co2 TEXT,
       created TIMESTAMP,
       notes TEXT,
       status INT,
PRIMARY KEY ((device_id)));

CREATE TABLE IF NOT EXISTS visions (
       device_id TEXT,
       objects TEXT,
       persons INT,
       gender TEXT,
       emotion TEXT,
       pose TEXT,
       action TEXT,
       created TIMESTAMP,
       notes TEXT,
       status INT,
PRIMARY KEY ((device_id)));

CREATE TABLE IF NOT EXISTS posenets (
       device_id TEXT,
       persons INT,
       NOSE TEXT,
       LEFT_EYE TEXT,
       RIGHT_EYE TEXT,
       LEFT_EAR TEXT,
       RIGHT_EAR TEXT,
       LEFT_SHOULDER TEXT,
       RIGHT_SHOULDER TEXT,
       LEFT_ELBOW TEXT,
       RIGHT_ELBOW TEXT,
       LEFT_WRIST TEXT,
       RIGHT_WRIST TEXT,
       LEFT_HIP TEXT,
       RIGHT_HIP TEXT,
       LEFT_KNEE TEXT,
       RIGHT_KNEE TEXT,
       LEFT_ANKLE TEXT,
       RIGHT_ANKLE TEXT,
       created TIMESTAMP,
       notes TEXT,
       status INT,
PRIMARY KEY ((device_id)));

CREATE TABLE IF NOT EXISTS attendance_rules (
       attendance_rules_id UUID,
       check_in TEXT,
       chech_out TEXT,
       break_start TEXT,
       break_finish TEXT,
       overtime_start TEXT,
       overtime_finish TEXT,
       overtime_on_holiday_start TEXT,
       overtime_on_holiday_finish TEXT,
       created TIMESTAMP,
       last_update TIMESTAMP,
       notes TEXT,
       status INT,
PRIMARY KEY ((attendance_rules_id)));

CREATE TABLE IF NOT EXISTS audios (
       attache_id TEXT,
       audio BLOB,
       transcription TEXT,
       created TIMESTAMP,
       last_edited TIMESTAMP,
       notes TEXT,
       status INT,
       audio_id UUID,
PRIMARY KEY ((audio_id), attache_id));
