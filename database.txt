# users

CREATE TABLE IF NOT EXISTS users (
       email TEXT,
       phone TEXT,
       fullname TEXT,
       nickname TEXT,
       password TEXT,
       device_id TEXT,
       device_status INT,
       attache_locale TEXT,
       attache_gender TEXT,
       attache_status INT,
       created TIMESTAMP,
       last_edited TIMESTAMP,
       status INT,
       user_id UUID,
PRIMARY KEY ((email), user_id))
WITH CLUSTERING ORDER BY (user_id DESC);
